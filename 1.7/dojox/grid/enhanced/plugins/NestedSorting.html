dojo.require("dojox.grid.enhanced.plugins.NestedSorting");
Defined in dojox/grid/enhanced/plugins/NestedSorting.js

Usage:

var foo = new dojox.grid.enhanced.plugins.NestedSorting(/* */ inGrid, /* */ option);

A flexible way to control multiple column sorting, including 1. Set default sorting order 2. Disable sorting for certain columns 3. Set sorting order dynamically with JS API

Examples:

Example 1

	<script type="text/javascript">
		var grid = new dojox.grid.EnhancedGrid({plugins : {nestedSorting: true}},
	               sortFields: [{attribute: 'col4', descending: false},...],//set default sorting order
			       canSort: function(index, field){ return true},//disable sorting for a column
				   ... }, dojo.byId('gridDiv'));
		grid.startup();
		//set new sorting order
		grid.setSortIndex([{attribute: 'col3', descending: true},...])
	</script>
